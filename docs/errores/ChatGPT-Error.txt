import csv
import json
import subprocess
import argparse
from pathlib import Path
from collections import defaultdict

SEPARATOR = "|"

def split_values(value):
    if not value:
        return []
    return [v.strip() for v in value.split(SEPARATOR) if v.strip()]

def run_sf_query(org, query):
    cmd = [
        "sf",
        "data",
        "query",
        "-q",
        query,
        "-o",
        org,
        "--json"
    ]
    result = subprocess.run(cmd, capture_output=True, text=True)

    if result.returncode != 0:
        raise RuntimeError(
            f"Error ejecutando query:\n{query}\n\nSTDERR:\n{result.stderr}"
        )

    try:
        data = json.loads(result.stdout)
    except json.JSONDecodeError as e:
        raise RuntimeError(
            f"No se pudo interpretar la respuesta JSON.\n\nSTDOUT:\n{result.stdout}\n\nSTDERR:\n{result.stderr}"
        ) from e

    if "result" not in data or "records" not in data["result"]:
        raise RuntimeError(
            f"Respuesta inesperada del comando sf.\n\n{json.dumps(data, indent=2)}"
        )

    return data["result"]["records"]

def write_csv(path, rows, headers):
    with open(path, "w", newline="", encoding="utf8") as f:
        writer = csv.DictWriter(
            f,
            fieldnames=headers,
            lineterminator="\n"
        )
        writer.writeheader()
        for r in rows:
            writer.writerow(r)

def load_master(master_file):
    matrix = defaultdict(lambda: {
        "psg": set(),
        "puesto": None,
        "segmentos": set(),
        "queues": set()
    })

    with open(master_file, newline="", encoding="utf8") as f:
        reader = csv.DictReader(f)

        for row in reader:
            role = row["RoleName"].strip()

            if not role:
                continue

            for psg in split_values(row.get("PermissionSetGroup", "")):
                matrix[role]["psg"].add(psg)

            puesto = (row.get("PublicGroupPuesto") or "").strip()
            if puesto:
                matrix[role]["puesto"] = puesto

            for seg in split_values(row.get("PublicGroupSegmento", "")):
                matrix[role]["segmentos"].add(seg)

            for q in split_values(row.get("Queues", "")):
                matrix[role]["queues"].add(q)

    return matrix

def load_target_users(path):
    users = []
    with open(path, newline="", encoding="utf8") as f:
        reader = csv.DictReader(f)
        for r in reader:
            users.append(r)
    return users

Sí 👍 pero con una aclaración importante en Salesforce:


Con SOQL no puedes eliminar registros.
SOQL solo consulta. Para quitar Permission Set Groups debes:


1️⃣ Consultar los PermissionSetAssignment del usuario
2️⃣ Eliminar esos registros



1️⃣ Query para ver qué Permission Set Groups tienen los usuarios "dummy"


SELECT 
    Id,
    Assignee.Username,
    PermissionSetGroupId,
    PermissionSetGroup.DeveloperName
FROM PermissionSetAssignment
WHERE Assignee.Username LIKE '%dummy%'
AND PermissionSetGroupId != NULL


Esto te mostrará exactamente qué PSG tiene cada usuario dummy.



2️⃣ Exportar esos registros con sf CLI


sf data query \
-q "SELECT Id FROM PermissionSetAssignment WHERE Assignee.Username LIKE '%dummy%' AND PermissionSetGroupId != NULL" \
-o TU_ORG \
-r csv > remove_psg.csv


El archivo quedará algo así:


Id
0PaXXXXXXXXXXXX
0PaXXXXXXXXXXXX
0PaXXXXXXXXXXXX



3️⃣ Eliminar los PSG (Bulk)


sf data delete bulk \
-s PermissionSetAssignment \
-f remove_psg.csv \
-o TU_ORG


Esto remueve los Permission Set Groups de todos los usuarios dummy.



🚀 Alternativa en un solo comando (más rápido)


sf data delete bulk \
-s PermissionSetAssignment \
-w "Assignee.Username LIKE '%dummy%' AND PermissionSetGroupId != NULL" \
-o TU_ORG



⚠️ Importante


Esto solo elimina Permission Set Groups, no los Permission Sets normales.


Si también quisieras eliminar Permission Sets directos, el query sería:


SELECT Id
FROM PermissionSetAssignment
WHERE Assignee.Username LIKE '%dummy%'
AND PermissionSetId != NULL
AND PermissionSetGroupId = NULL

Para quitar la asignación de un Permission Set Group (PSG) no se hace “editando” el usuario ni con un UPDATE. En Salesforce, la asignación vive en el objeto PermissionSetAssignment, y para removerla hay que borrar ese registro. Salesforce indica que PermissionSetAssignment representa la asignación de un permission set o permission set group a un usuario, y que para quitar una asignación se usa delete. Developer+1


Hazlo así:


1. Consultar qué asignaciones vas a borrar


Para tu caso de “todos los dummy excepto esos 5”:


SELECT 
    Id,
    Assignee.Username,
    PermissionSetGroup.DeveloperName
FROM PermissionSetAssignment
WHERE Assignee.Username LIKE '%dummy%'
AND Assignee.Username NOT IN (
    'admin.dummy@telecentro.ams',
    'gestion_administrativa.dummy',
    'alianzas.dummy',
    'gerencial_starlink.dummy',
    'dirección_administrativa.dummy'
)
AND PermissionSetGroupId != NULL
ORDER BY Assignee.Username


Ese query te muestra exactamente qué PSG se le quitará a cada usuario.


2. Exportar los IDs de PermissionSetAssignment


sf data query \
-q "SELECT Id FROM PermissionSetAssignment WHERE Assignee.Username LIKE '%dummy%' AND Assignee.Username NOT IN ('admin.dummy@telecentro.ams','gestion_administrativa.dummy','alianzas.dummy','gerencial_starlink.dummy','dirección_administrativa.dummy') AND PermissionSetGroupId != NULL" \
-o TU_ORG \
-r csv > remove_psg.csv

El comando de Salesforce CLI para borrado masivo existe en sf data delete bulk. Developer


sf data delete bulk \
-s PermissionSetAssignment \
-f remove_psg.csv \
-o TU_ORG


Qué pasa realmente


Ese borrado elimina las filas de PermissionSetAssignment, así que el usuario queda sin ese PSG. No borra el PSG del org ni toca roles, colas, grupos públicos o permission sets que no pertenezcan a un group. Developer+1

async function waitForAgentforceAgents(page, maxAttempts = 8, waitMs = 10000) {
  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    const page1 = await openSetup(page);

    try {
      await page1.getByRole('searchbox', { name: 'Quick Find' }).click();
      await page1.getByRole('searchbox', { name: 'Quick Find' }).fill('');
      await page1.getByRole('searchbox', { name: 'Quick Find' }).pressSequentially('Agentforce Agents');

      const agentforceLink = page1.getByRole('link', {
        name: 'Agentforce Agents',
        exact: true
      });

      await agentforceLink.waitFor({ state: 'visible', timeout: 5000 });
      await agentforceLink.click();

      return page1;
    } catch (e) {
      await page1.close();

      if (attempt < maxAttempts) {
        await page.bringToFront();
        await page.waitForTimeout(waitMs);
      }
    }
  }

  throw new Error('Agentforce Agents no apareció en Quick Find');
}

  await page.goto(`${baseUrl}/lightning/page/home`);

  // abrir menú de setup
  await page.getByRole('button', { name: 'Setup' }).click();

  // intentar abrir el popup de Setup usando varios selectores posibles
  const popupPromise = page.waitForEvent('popup', { timeout: 30000 });

  const setupOptionCandidates = [
    page.getByRole('menuitem', { name: 'Setup Opens in a new tab Setup for current app' }),
    page.getByRole('menuitem', { name: /^Setup$/ }),
    page.getByRole('link', { name: /^Setup$/ }),
    page.getByText('Setup', { exact: true }).last()
  ];

  let clicked = false;

  for (const candidate of setupOptionCandidates) {
    try {
      await candidate.waitFor({ state: 'visible', timeout: 5000 });
      await candidate.click();
      clicked = true;
      break;
    } catch (e) {
      // probar siguiente opción
    }
  }

  if (!clicked) {
    throw new Error('No se pudo encontrar la opción Setup dentro del menú de Setup');
  }

  const popup = await popupPromise;
  await popup.waitForLoadState('domcontentloaded');

  return popup;
}
